180  // NOTE: Change core voltage one level at a time..
181  SetVcoreUp (0x01);
182  SetVcoreUp (0x02);
183  SetVcoreUp (0x03);

187  UCSCTL6 &= ~XT2BYPASS;

189  UCSCTL4 |= SELA_2 + SELS__DCOCLKDIV + SELM__DCOCLKDIV;
190 
191  UCSCTL0 = 0x0000; // Set lowest possible DCOx, MODx
192  // Loop until XT1,XT2 & DCO stabilizes - In this case only DCO has to stabilize
193  do

197  SFRIFG1 &= ~OFIFG; // Clear fault flags

201  __bis_SR_register(SCG0);

231  // Set SVS/SVM high side new level
