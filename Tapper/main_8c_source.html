47 void SetClk24MHz(){
48  // Increase Vcore setting to level3 to support fsystem=25MHz
49  // NOTE: Change core voltage one level at a time..
50  SetVcoreUp (0x01);
51  SetVcoreUp (0x02);
52  SetVcoreUp (0x03);
53 
54  P5SEL |= BIT2+BIT3;
55  UCSCTL6 &= ~XT2OFF; // Enable XT2
56  UCSCTL6 &= ~XT2BYPASS;
57  UCSCTL3 = SELREF__XT2CLK; // FLLref = XT2
58  UCSCTL4 |= SELA_2 + SELS__DCOCLKDIV + SELM__DCOCLKDIV;
59 
60  UCSCTL0 = 0x0000; // Set lowest possible DCOx, MODx
61  // Loop until XT1,XT2 & DCO stabilizes - In this case only DCO has to stabilize
62  do
63  {
64  UCSCTL7 &= ~(XT2OFFG + XT1LFOFFG + DCOFFG);
65  // Clear XT2,XT1,DCO fault flags
66  SFRIFG1 &= ~OFIFG; // Clear fault flags
67  }while (SFRIFG1&OFIFG); // Test oscillator fault flag
68 
69  // Disable the FLL control loop
70  __bis_SR_register(SCG0);
71 
72  // Select DCO range 24MHz operation
73  UCSCTL1 = DCORSEL_7;
74  /* Set DCO Multiplier for 24MHz
75  (N + 1) * FLLRef = Fdco
76  (5 + 1) * 4MHz = 24MHz */
77  UCSCTL2 = FLLD0 + FLLN0 + FLLN2;
78  // Enable the FLL control loop
79  __bic_SR_register(SCG0);
80 
81  /* Worst-case settling time for the DCO when the DCO range bits have been
82  changed is n x 32 x 32 x f_MCLK / f_FLL_reference. See UCS chapter in 5xx
83  UG for optimization.
84  32 x 32 x 24MHz / 4MHz = 6144 = MCLK cycles for DCO to settle */
85  __delay_cycles(70000);
86 
87  // Loop until XT1,XT2 & DCO stabilizes - In this case only DCO has to stabilize
88  do {
89  // Clear XT2,XT1,DCO fault flags
90  UCSCTL7 &= ~(XT2OFFG + XT1LFOFFG + DCOFFG);
91  // Clear fault flags
92  SFRIFG1 &= ~OFIFG;
93  } while (SFRIFG1 & OFIFG); // Test oscillator fault flag

96 void SetVcoreUp (unsigned int level)
97 {
98  // Open PMM registers for write
99  PMMCTL0_H = PMMPW_H;
100  // Set SVS/SVM high side new level
101  SVSMHCTL = SVSHE + SVSHRVL0 * level + SVMHE + SVSMHRRL0 * level;
102  // Set SVM low side to new level
103  SVSMLCTL = SVSLE + SVMLE + SVSMLRRL0 * level;
104  // Wait till SVM is settled
105  while ((PMMIFG & SVSMLDLYIFG) == 0);
106  // Clear already set flags
107  PMMIFG &= ~(SVMLVLRIFG + SVMLIFG);
108  // Set VCore to new level
109  PMMCTL0_L = PMMCOREV0 * level;
110  // Wait till new level reached
111  if ((PMMIFG & SVMLIFG))
112  while ((PMMIFG & SVMLVLRIFG) == 0);
113  // Set SVS/SVM low side to new level
114  SVSMLCTL = SVSLE + SVSLRVL0 * level + SVMLE + SVSMLRRL0 * level;
115  // Lock PMM registers for write access
116  PMMCTL0_H = 0x00;
117 }
