172 void SetClk24MHz(){
173  // Increase Vcore setting to level3 to support fsystem=25MHz
174  // NOTE: Change core voltage one level at a time..
175  SetVcoreUp (0x01);
176  SetVcoreUp (0x02);
177  SetVcoreUp (0x03);
178 
179  P5SEL |= BIT2+BIT3;
180  UCSCTL6 &= ~XT2OFF; // Enable XT2
181  UCSCTL6 &= ~XT2BYPASS;
182  UCSCTL3 = SELREF__XT2CLK; // FLLref = XT2
183  UCSCTL4 |= SELA_2 + SELS__DCOCLKDIV + SELM__DCOCLKDIV;
184 
185  UCSCTL0 = 0x0000; // Set lowest possible DCOx, MODx
186  // Loop until XT1,XT2 & DCO stabilizes - In this case only DCO has to stabilize
187  do
188  {
189  UCSCTL7 &= ~(XT2OFFG + XT1LFOFFG + DCOFFG);
190  // Clear XT2,XT1,DCO fault flags
191  SFRIFG1 &= ~OFIFG; // Clear fault flags
192  }while (SFRIFG1&OFIFG); // Test oscillator fault flag
193 
194  // Disable the FLL control loop
195  __bis_SR_register(SCG0);
196 
197  // Select DCO range 24MHz operation
198  UCSCTL1 = DCORSEL_7;
199  /* Set DCO Multiplier for 24MHz
200  (N + 1) * FLLRef = Fdco
201  (5 + 1) * 4MHz = 24MHz */
202  UCSCTL2 = FLLD0 + FLLN0 + FLLN2;
203  // Enable the FLL control loop
204  __bic_SR_register(SCG0);
205 
206  /* Worst-case settling time for the DCO when the DCO range bits have been
207  changed is n x 32 x 32 x f_MCLK / f_FLL_reference. See UCS chapter in 5xx
208  UG for optimization.
209  32 x 32 x 24MHz / 4MHz = 6144 = MCLK cycles for DCO to settle */
210  __delay_cycles(70000);
211 
212  // Loop until XT1,XT2 & DCO stabilizes - In this case only DCO has to stabilize
213  do {
214  // Clear XT2,XT1,DCO fault flags
215  UCSCTL7 &= ~(XT2OFFG + XT1LFOFFG + DCOFFG);
216  // Clear fault flags
217  SFRIFG1 &= ~OFIFG;
218  } while (SFRIFG1 & OFIFG); // Test oscillator fault flag
219 }

221 void SetVcoreUp (unsigned int level)
222 {
223  // Open PMM registers for write
224  PMMCTL0_H = PMMPW_H;
225  // Set SVS/SVM high side new level
226  SVSMHCTL = SVSHE + SVSHRVL0 * level + SVMHE + SVSMHRRL0 * level;
227  // Set SVM low side to new level
228  SVSMLCTL = SVSLE + SVMLE + SVSMLRRL0 * level;
229  // Wait till SVM is settled
230  while ((PMMIFG & SVSMLDLYIFG) == 0);
231  // Clear already set flags
232  PMMIFG &= ~(SVMLVLRIFG + SVMLIFG);
233  // Set VCore to new level
234  PMMCTL0_L = PMMCOREV0 * level;
235  // Wait till new level reached
236  if ((PMMIFG & SVMLIFG))
237  while ((PMMIFG & SVMLVLRIFG) == 0);
238  // Set SVS/SVM low side to new level
239  SVSMLCTL = SVSLE + SVSLRVL0 * level + SVMLE + SVSMLRRL0 * level;
240  // Lock PMM registers for write access
241  PMMCTL0_H = 0x00;
242 }
