193 void Serve(void) {
194  static uint32_t last_served;
195  volatile uint8_t i;
196  // pointer to glass to use
197  char_object_t * full = 0;
198  // make sure a glass was not just served
199  if (TimeSince(last_served) <= SERVE_SPEED) return;
200 
201  // find an unused glass
202  for(i = 0; i < MAX_GLASSES; i++) if(FULL_GLASSES[i].status == 0) full = &FULL_GLASSES[i];
203  if(full && game.glasses) {
204  // schedule MoveUpShot1 to run every SERVE_SPEED ms, this is what makes the 't' move
205  Task_Schedule((task_t)MoveLeftGlass, full, SERVE_SPEED, SERVE_SPEED);
206  game.glasses--;
207  }else { // if no fulls are left
208  // indicate out of ammo by changing the pointer
209  if (game.c == 't') {
210  game.c = 'i';
211  Game_CharXY(game.c, MAP_WIDTH - 1, game.y);
212  }
213  return;
214  }
215  // change the status of the shot to indicate that it is used
216  full->status = 1;
217  // start the glass at the right just to the left of the bartender
218  full->x = MAP_WIDTH - 2;
219  full->y = game.y;
220  full->c = 'U';
221  Game_CharXY('t', MAP_WIDTH - 2, game.y);
222  // track the time the last shot was fired so you can't have two on top of each other
223  last_served = TimeNow();
224  UpdateGlasses();
225 }

253  // find unused patron object

257  break;

260  break;
261  case 5:
262  for(i = 0; i < MAX_PATRONS; i++) if(patrons3[i].status == 0) patron = &patrons3[i];
263  break;
264  }
265  if(patron) {
266  patron->status = 1;
267  patron->x = 1;
268  patron->c = 'o';
269  patron->y = y;
270  // use the task scheduler to move the char right at the calculated period
271  Task_Schedule((task_t)MoveRightPatron, patron, right_period, right_period);
272  Game_CharXY('o', 1, y);
273  }
274  // schedule next patron to come between min and max patron rate
275  // since a random time is desired the period is 0 so the task will be done when it us run
276  // and a new task will be created at a different random time
277  Task_Schedule(SendPatron, 0, random_int(MIN_PATRON_RATE/2, MAX_PATRON_RATE/2), 0);
